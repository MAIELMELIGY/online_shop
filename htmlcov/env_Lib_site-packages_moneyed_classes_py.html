28class Currency(object): 

29 """ 

30 A Currency represents a form of money issued by governments, and 

31 used in one or more states/countries. A Currency instance 

32 encapsulates the related data of: the ISO currency/numeric code, a 

33 canonical name, and countries the currency is used in. 

34 """ 

35

38 self.numeric = numeric 

39 self.sub_unit = sub_unit 

40 if name is not None: 

41 self.name = name 

42 if countries is not None: 

43 self.countries = countries

127class Money(object): 

128 """ 

129 A Money instance is a combination of data - an amount and a 

130 currency - along with operators that handle the semantics of money 

131 operations in a better way than just dealing with raw Decimal or 

132 ($DEITY forbid) floats. 

133 """ 

134 

135 def __init__(self, amount=Decimal('0.0'), currency=DEFAULT_CURRENCY_CODE): 

136 if not isinstance(amount, Decimal): 

137 amount = Decimal(str(amount)) 

138 self.amount = amount 

139 

140 if not isinstance(currency, Currency): 

141 currency = get_currency(str(currency).upper()) 

142 self.currency = currency 

143 

144 def __repr__(self): 

145 return "Money('%s', '%s')" % (self.amount, self.currency) 

146 

147 def __unicode__(self): 

148 return format_money(self) 

149 

150 if PYTHON2: 

151 def __str__(self): 

152 # On Python 2, `__str__` returns byte strings, so we can't include unicode symbols. 

153 # Use a simpler fallback that avoids format_money 

154 return '{0}{1:,}'.format(self.currency.code, self.amount) 

155 

156 else: 

157 def __str__(self): 

158 return format_money(self) 

159 

160 def __hash__(self): 

161 return hash((self.amount, self.currency)) 

162 

163 def __pos__(self): 

164 return self.__class__( 

165 amount=self.amount, 

166 currency=self.currency) 

167 

168 def __neg__(self): 

169 return self.__class__( 

170 amount=-self.amount, 

171 currency=self.currency) 

172 

173 def __add__(self, other): 

174 if other == 0: 

175 # This allows things like 'sum' to work on list of Money instances, 

176 # just like list of Decimal. 

177 return self 

178 if not isinstance(other, Money): 

179 return NotImplemented 

180 if self.currency == other.currency: 

181 return self.__class__( 

182 amount=self.amount + other.amount, 

183 currency=self.currency) 

184 

185 raise TypeError('Cannot add or subtract two Money ' + 

186 'instances with different currencies.') 

187 

188 def __sub__(self, other): 

189 return self.__add__(-other) 

190 

191 def __rsub__(self, other): 

192 return (-self).__add__(other) 

193 

194 def __mul__(self, other): 

195 if isinstance(other, Money): 

196 raise TypeError('Cannot multiply two Money instances.') 

197 else: 

198 if isinstance(other, float): 

199 warnings.warn("Multiplying Money instances with floats is deprecated", DeprecationWarning) 

200 return self.__class__( 

201 amount=(self.amount * force_decimal(other)), 

202 currency=self.currency) 

203 

204 def __truediv__(self, other): 

205 if isinstance(other, Money): 

206 if self.currency != other.currency: 

207 raise TypeError('Cannot divide two different currencies.') 

208 return self.amount / other.amount 

209 else: 

210 if isinstance(other, float): 

211 warnings.warn("Dividing Money instances by floats is deprecated", DeprecationWarning) 

212 return self.__class__( 

213 amount=(self.amount / force_decimal(other)), 

214 currency=self.currency) 

215 

216 def __rtruediv__(self, other): 

217 raise TypeError('Cannot divide non-Money by a Money instance.') 

218 

219 def round(self, ndigits=0): 

220 """ 

221 Rounds the amount using the current ``Decimal`` rounding algorithm. 

222 """ 

223 if ndigits is None: 

224 ndigits = 0 

225 return self.__class__( 

226 amount=self.amount.quantize(Decimal('1e' + str(-ndigits))), 

227 currency=self.currency) 

228 

229 def __abs__(self): 

230 return self.__class__( 

231 amount=abs(self.amount), 

232 currency=self.currency) 

233 

234 def __bool__(self): 

235 return bool(self.amount) 

236 

237 if PYTHON2: 

238 __nonzero__ = __bool__

240 def __rmod__(self, other): 

241 """ 

242 Calculate percentage of an amount. The left-hand side of the 

243 operator must be a numeric value. 

244 

245 Example: 

246 >>> money = Money(200, 'USD') 

247 >>> 5 % money 

248 USD 10.00 

249 """ 

250 if isinstance(other, Money): 

251 raise TypeError('Invalid __rmod__ operation') 

252 else: 

253 if isinstance(other, float): 

254 warnings.warn("Calculating percentages of Money instances using floats is deprecated", 

255 DeprecationWarning) 

256 return self.__class__( 

257 amount=(Decimal(str(other)) * self.amount / 100), 

258 currency=self.currency) 

259 

260 __radd__ = __add__ 

261 __rmul__ = __mul__

269

276 raise MoneyComparisonError(other)

283 if not isinstance(other, Money):

286 return (self.amount > other.amount)

308def add_currency(code, numeric, sub_unit=1, name=None, countries=None): 

309 currency = Currency(code=code, numeric=numeric, sub_unit=sub_unit, name=name, countries=countries) 

310 CURRENCIES[code] = currency 

311 # No lookup by numeric code for currencies without numeric codes 

312 if numeric is not None:

317def get_currency(code=None, iso=None): 

318 try:

322 except KeyError: 

323 raise CurrencyDoesNotExist(code)

25 def add_sign_definition(self, locale, currency, prefix='', suffix=''): 

26 locale = locale.upper() 

27 currency_code = currency.code.upper() 

28 if locale not in self.sign_definitions: 

29 self.sign_definitions[locale] = {} 

30 self.sign_definitions[locale][currency_code] = (prefix, suffix)

32 def add_formatting_definition(self, locale, group_size, 

33 group_separator, decimal_point, 

34 positive_sign, trailing_positive_sign, 

35 negative_sign, trailing_negative_sign, 

36 rounding_method): 

37 locale = locale.upper() 

38 self.formatting_definitions[locale] = { 

39 'group_size': group_size, 

40 'group_separator': group_separator, 

41 'decimal_point': decimal_point, 

42 'positive_sign': positive_sign, 

43 'trailing_positive_sign': trailing_positive_sign, 

44 'negative_sign': negative_sign, 

45 'trailing_negative_sign': trailing_negative_sign, 

46 'rounding_method': rounding_method}

48 def get_sign_definition(self, currency_code, locale): 

49 currency_code = currency_code.upper() 

50 

51 if locale.upper() not in self.sign_definitions: 

52 locale = DEFAULT 

53 

54 local_set = self.sign_definitions.get(locale.upper()) 

55 

56 if currency_code in local_set: 

57 return local_set.get(currency_code) 

58 else: 

59 ret = self.sign_definitions.get(DEFAULT).get(currency_code) 

60 return ret if ret is not None else ('', " %s" % currency_code) 

61 

62 def get_formatting_definition(self, locale): 

63 if locale.upper() not in self.formatting_definitions: 

64 locale = DEFAULT 

65 return self.formatting_definitions.get(locale.upper())

67 def format(self, money, include_symbol=True, locale=DEFAULT, 

68 decimal_places=None, rounding_method=None): 

69 locale = locale.upper() 

70 code = money.currency.code.upper() 

71 prefix, suffix = self.get_sign_definition(code, locale) 

72 formatting = self.get_formatting_definition(locale) 

73 

74 if rounding_method is None: 

75 rounding_method = formatting['rounding_method'] 

76 

77 if decimal_places is None: 

78 # TODO: Use individual defaults for each currency 

79 decimal_places = 2 

80 

81 q = Decimal(10) ** -decimal_places # 2 places --> '0.01' 

82 quantized = money.amount.quantize(q, rounding_method) 

83 negative, digits, e = quantized.as_tuple() 

84 

85 result = [] 

86 

87 digits = list(map(str, digits)) 

88 

89 build, next = result.append, digits.pop 

90 

91 # Trailing sign 

92 if negative: 

93 build(formatting['trailing_negative_sign']) 

94 else: 

95 build(formatting['trailing_positive_sign']) 

96 

97 # Suffix 

98 if include_symbol: 

99 build(suffix) 

100 

101 # Decimals 

102 for i in range(decimal_places): 

103 build(next() if digits else '0') 

104 

105 # Decimal points 

106 if decimal_places: 

107 build(formatting['decimal_point']) 

108 

109 # Grouped number 

110 if not digits: 

111 build('0') 

112 else: 

113 i = 0 

114 while digits: 

115 build(next()) 

116 i += 1 

117 if i == formatting['group_size'] and digits: 

118 i = 0 

119 build(formatting['group_separator']) 

120 

121 # Prefix 

122 if include_symbol: 

123 build(prefix) 

124 

125 # Sign 

126 if negative: 

127 build(formatting['negative_sign']) 

128 else: 

129 build(formatting['positive_sign']) 

130 

131 return ''.join(reversed(result))
